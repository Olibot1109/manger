# ── Manger Permissions ────────────────────────────────────────────────────────
#
# Add permission names to allowedActions or deniedActions in accounts.json.
# Admins (admin: true) bypass all permission checks entirely.
#
#   mode "allow"  →  user can ONLY do actions listed in allowedActions
#   mode "deny"   →  user can do EVERYTHING except actions in deniedActions
#
# ── Per-client actions ────────────────────────────────────────────────────────

ban            # ban a specific client
unban          # unban a specific client
toggleban      # ban/unban toggle button (requires ban + unban to fully allow)
delete         # delete a client record permanently
redirect       # redirect a single client to a URL
message        # send a popup message to a single client
image          # send an image overlay to a single client
effect         # apply a visual effect to a client (invert, blur, etc.)
notes          # set or edit the private note on a client
timeout        # put a client in timed timeout
untimeout      # remove a client's active timeout early
reload         # force-reload a client's page
close          # force-close a client's tab
cooke          # clear a client's cookies
cooke_add      # inject custom cookies into a client

# ── Bulk actions ──────────────────────────────────────────────────────────────

deleteAll      # wipe all client records
redirectAll    # redirect every active client to a URL
messageAll     # send a message to every active client
closeAll       # close every active client's tab
reloadAll      # reload every active client's page
showIdAll      # broadcast client IDs to all clients
sendImageAll   # send an image to every active client

# ── System ────────────────────────────────────────────────────────────────────

lockdown       # enable / disable lockdown mode (forces all clients to a URL)
poll           # create, activate, close, and delete polls
